            return word.toLowerCase().split('').map((val, index, arr) => {
                return arr.indexOf(val) === arr.lastIndexOf(val) ? '(' : ')'
            }).join('');
        }
        
        function findOdd(A) {
            let sortedArr = A.sort();
            let index = 0;
            let resArr = [[sortedArr[index], 1]];
            for(let i = 1; i < sortedArr.length; i++) {
                if(sortedArr[i] === sortedArr[i-1]) {
                resArr[index][1]++;
                } else {
                if (resArr[index][1] % 2 === 1) return resArr[index][0];
                
                resArr.push([sortedArr[i],1]);
                index++;
                }
            }
            
            return resArr[index][0];
        }
        
        function findOutlier(integers){
            let res = 0;
            for(let i = 0; i < 3; i++) {
                res += (Math.abs(integers[i]) % 2);
            }
            for(let j = 0; j < integers.length; j++) {
                if(res > 1 && integers[j] % 2 === 0) {
                return integers[j];
                } else if(res <= 1 && integers[j] % 2 !== 0) {
                return integers[j];
                }
            }
        }
        
        function fridayTheThirteenths(start, end) {
            let result = [];
            end = end ? end : start;
            for(let year = start; year <= end; year++) {
                for(let month = 0; month < 12; month++) {
                let checkDate = new Date(year, month, 13);
                if(checkDate.getDay() === 5) result.push(`${month+1}/13/${year}`);
                }
            }
            
            return result.join(' ');
            }
